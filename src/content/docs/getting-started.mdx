This guide walks the complete ConvoMem integration loop: capture a conversation, wait for
extraction, then recall context at the start of the next interaction.

## 2. Initialize

Get your API key from the [dashboard](https://app.convomem.com/settings/keys).

<CodeTabs>
  <CodeTab language="typescript" filename="client.ts">
  ```typescript
  import { ConvoMem } from '@optraze/convomem-sdk'

  const client = new ConvoMem({
    apiKey: process.env.CONVOMEM_API_KEY!,
  })
  ```
  </CodeTab>
  <CodeTab language="python" filename="client.py">
  ```python

  from convomem import ConvoMemClient

  client = ConvoMemClient(api_key=os.environ["CONVOMEM_API_KEY"])
  ```

  </CodeTab>
  <CodeTab language="rust" filename="main.rs">
  ```rust

  ```
  </CodeTab>
</CodeTabs>

## 3. Capture a conversation

Send conversation turns as they happen. ConvoMem resolves the customer from whatever
identity field you provide — no explicit customer creation needed.

<CodeTabs>
  <CodeTab language="typescript" filename="capture.ts">
  ```typescript
  await client.capture(
    [
      { role: 'user', content: 'Hi, I need help with my subscription. I am on the Pro plan.' },
      { role: 'assistant', content: 'Of course! I can see your Pro plan. What do you need?' },
    ],
    { email: 'alice@example.com' },
    { channel: 'CHAT' },
  )
  ```
  </CodeTab>
  <CodeTab language="python" filename="capture.py">
  ```python
  from convomem import CaptureRequest, Message

  client.capture(CaptureRequest(
      messages=[
          Message(role="user", content="Hi, I need help with my subscription. I am on the Pro plan."),
          Message(role="assistant", content="Of course! What do you need?"),
      ],
      email="alice@example.com",
      channel="CHAT",
  ))
  ```
  </CodeTab>
  <CodeTab language="rust" filename="main.rs">
  ```rust
  use convomem::{CaptureRequest, Message};

  client.capture(&CaptureRequest {
      messages: Some(vec![
          Message { role: "user".into(), content: "Hi, I need help with my subscription. I am on the Pro plan.".into() },
          Message { role: "assistant".into(), content: "Of course! What do you need?".into() },
      ]),
      email: Some("alice@example.com".into()),
      channel: Some("CHAT".into()),
      ..Default::default()
  }).await?;
  ```
  </CodeTab>

## 4. Recall context

Memory extraction happens in the background. A few seconds later — or at the start of the
next conversation — look up the relevant context:

<CodeTabs>
  <CodeTab language="typescript" filename="lookup.ts">
  ```typescript
  const { context, memories } = await client.lookup(
    'subscription and plan details',
    { email: 'alice@example.com' },
  )

  console.log(context)
  // → "Alice is on the Pro plan."
  console.log(`${memories.length} memories found`)
  ```
  </CodeTab>
  <CodeTab language="python" filename="lookup.py">
  ```python
  result = client.memories.lookup("alice@example.com", "subscription and plan details")

  print(f"Context: {result.context}")
  print(f"Memories: {len(result.memories)}")
  ```
  </CodeTab>
  <CodeTab language="rust" filename="main.rs">
  ```rust
  let result = client.lookup_memories("alice@example.com", "subscription and plan details").await?;

  println!("Context: {}", result.context);
  println!("Memories: {}", result.memories.len());
  ```
  </CodeTab>
  <CodeTab language="curl" filename="Terminal">
  ```bash
  curl -s -H "X-API-Key: $CONVOMEM_API_KEY" \
    "https://api.convomem.com/api/v1/customers/memories/lookup?email=alice%40example.com&topic=subscription+and+plan+details"
  ```
  </CodeTab>
</CodeTabs>

## 5. Inject context into your LLM

Drop `context` into the system prompt before every interaction:

```typescript title="agent.ts"
import OpenAI from 'openai'
import { ConvoMem } from '@optraze/convomem-sdk'

const client = new ConvoMem({ apiKey: process.env.CONVOMEM_API_KEY! })
const openai = new OpenAI()

async function chat(userEmail: string, userMessage: string) {
  // 1. Recall what we know
  const { context } = await client.lookup(userMessage, { email: userEmail })

  // 2. Build prompt with memory
  const reply = await openai.chat.completions.create({
    model: 'gpt-4o',
    messages: [
      {
        role: 'system',
        content: `You are a helpful support agent.\n\nWhat you know about this customer:\n${context}`,
      },
      { role: 'user', content: userMessage },
    ],
  })

  // 3. Capture the exchange
  await client.capture(
    [
      { role: 'user', content: userMessage },
      { role: 'assistant', content: reply.choices[0].message.content! },
    ],
    { email: userEmail },
  )

  return reply.choices[0].message.content
}
```

## The agent loop

```
start of session:  lookup(topic, identity)    → inject context into system prompt
user sends turn:   capture([turns], identity) → queue memory extraction
repeat each turn
end of session:    endConversation(id, identity, { outcome })
```

<Callout type="tip" title="Always recall before capture">
Look up context at the start of the session, then capture as the conversation unfolds.
Captured memories take a few seconds to become searchable — don't look up immediately
after capture.
</Callout>
